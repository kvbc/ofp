\begin{center} \DoxyHorRuler{0}
 \end{center} \hypertarget{installation_autotoc_md2}{}\doxysection{Compiling from source}\label{installation_autotoc_md2}
First install the latest source distribution from \href{https://github.com/kvbc/ofp}{\texttt{ github}}, if you haven\textquotesingle{}t already \href{https://github.com/kvbc/ofp}{\texttt{ }}


\begin{DoxyEnumerate}
\item Unpack the archive
\begin{DoxyItemize}
\item {\bfseries{on Linux}} \begin{quote}
{\ttfamily gunzip ofp-\/\+VERSION\+\_\+src.\+tar.\+gz} ~\newline
 {\ttfamily tar xf ofp-\/\+VERSION\+\_\+src.\+tar} \end{quote}

\item {\bfseries{on Windows}} \begin{quote}
{\ttfamily tar zxvf ofp-\/\+VERSION\+\_\+src.\+tar.\+gz} \end{quote}

\end{DoxyItemize}
\item Configure the {\ttfamily makefile} to suit your needs and run {\ttfamily make} to compile a
\begin{DoxyItemize}
\item {\bfseries{Static library}} \begin{quote}
{\ttfamily make o} ~\newline
 {\ttfamily make slib} \end{quote}

\item {\bfseries{Dynamic library}} \begin{quote}
{\ttfamily make o} ~\newline
 {\ttfamily make dlib} \end{quote}

\end{DoxyItemize}
\item Clean the directory \begin{quote}
{\ttfamily make clean} \end{quote}
